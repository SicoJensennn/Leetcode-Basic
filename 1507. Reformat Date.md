### 1507. Reformat Date

```python
class Solution:
    def reformatDate(self, date: str) -> str:
        date = date.split(" ")
        month = {"Jan":"01", "Feb":"02", "Mar":"03", "Apr":"04", "May":"05", "Jun":"06", "Jul":"07", "Aug":"08", "Sep":"09", "Oct":"10", "Nov":"11", "Dec":"12"}
        
        date[0] = date[0][: -2].zfill(2)
        date[1] = month.get(date[1])
        date.reverse()
        
        return "-".join(date)
```
